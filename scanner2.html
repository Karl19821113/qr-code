    function playBeep() {
    try {
    var ac = new (window.AudioContext || window.webkitAudioContext)();
    var osc = ac.createOscillator();
    var gain = ac.createGain();
    osc.connect(gain);
    gain.connect(ac.destination);
    osc.type = 'sine';
    osc.frequency.value = 880;
    gain.gain.value = 0.3;

    function loadParticipantData() {
    if (!gasUrl || !pin) {
    statusBar.className = 'status-bar error';
    statusBar.textContent = '❌ 缺少 GAS URL 或密碼參數';
    return;
    }
    statusBar.className = 'status-bar loading';
    statusBar.textContent = '📚 正在載入學員資料庫...';
    var apiUrl = gasUrl + '?action=getData&pin=' + encodeURIComponent(pin);
    fetch(apiUrl, { method: 'GET', redirect: 'follow' })
    .then(function (resp) { return resp.text(); })
    .then(function (text) {
    try {
    var result = JSON.parse(text);
    if (result.success && result.data) {
    participantMap = {};
    var count = 0;
    // 載入學員
    if (result.data.students) {
    result.data.students.forEach(function (s) {
    participantMap[s.id] = s;
    count++;
    });
    }
    // 載入同工
    if (result.data.staff) {
    result.data.staff.forEach(function (s) {
    participantMap[s.id] = s;
    count++;
    });
    }
    isAdmin = result.isAdmin || false;
    dataLoaded = true;
    statusBar.className = 'status-bar waiting';
    statusBar.textContent = '✅ 已載入 ' + count + ' 筆資料，請掃描 QR Code';
    debugInfo.textContent = '資料庫: ' + count + ' 筆 | 權限: ' + (isAdmin ? '管理員' : '班長');
    // 啟動同步計時器
    startSyncTimer();
    } else {
    statusBar.className = 'status-bar error';
    statusBar.textContent = '❌ ' + (result.message || '載入失敗');
    }
    } catch (e) {
    statusBar.className = 'status-bar error';
    statusBar.textContent = '❌ 資料解析失敗';
    }
    })
    .catch(function (err) {
    statusBar.className = 'status-bar error';
    statusBar.textContent = '❌ 網路錯誤: ' + err.message;
    });
    }
    // ====== 2. 掃描結果處理 (本地即時驗證) ======
    function handleScanResult(scannedCode) {
    if (!dataLoaded) {
    statusBar.className = 'status-bar error';
    statusBar.textContent = '⏳ 資料庫尚未載入完成';
    return;
    }
    // 冷卻檢查
    var now = Date.now();
    if (scannedCode === lastSent && (now - lastSentTime) < COOLDOWN_MS) return;
    lastSent = scannedCode;
    lastSentTime = now;
    // 本地查詢
    var person = participantMap[scannedCode];
    if (!person) {
    playErrorBeep();
    statusBar.className = 'status-bar error';
    statusBar.textContent = '❌ 無效條碼: ' + scannedCode;
    resetStatusLater(2500);
    return;
    }
    // 權限檢查
    if (!isAdmin && person.type === '同工') {
    playErrorBeep();
    statusBar.className = 'status-bar error';
    statusBar.textContent = '⛔ 同工簽到需牧長權限';
    resetStatusLater(2500);
    return;
    }
    // ✅ 成功! 即時回饋
    playBeep();
    statusBar.className = 'status-bar success';
    statusBar.textContent = '✅ ' + person.name + ' (' + (person.group || person.course || '') + ') 簽到成功';
    // 加入同步佇列
    syncQueue.push({
    id: person.id,
    mode: scanMode,
    name: person.name,
    type: person.type,
    rowIndex: person.row,
    timestamp: now
    });
    // 更新最近紀錄 UI
    addRecentScan(person.name, false);
    updateQueueBar();
    resetStatusLater(2000);
    // 通知 opener (如果有)
    try {
    if (window.opener && !window.opener.closed) {
    window.opener.postMessage({
    type: 'qr-scan-result',
    code: scannedCode,
    timestamp: now
    }, '*');
    }
    } catch (e) { }
    }

    syncTimer = setInterval(syncPendingQueue, SYNC_INTERVAL);
    }
    function syncPendingQueue() {
    if (isSyncing || syncQueue.length === 0 || !gasUrl || !pin) return;
    isSyncing = true;
    var toSync = syncQueue.splice(0, syncQueue.length); // 取出所有待同步
    updateQueueBar();
    var apiUrl = gasUrl +
    '?action=batchScan' +
    '&pin=' + encodeURIComponent(pin) +
    '&records=' + encodeURIComponent(JSON.stringify(toSync));
    fetch(apiUrl, { method: 'GET', redirect: 'follow' })
    .then(function (resp) { return resp.text(); })
    .then(function (text) {
    isSyncing = false;
    try {
    var result = JSON.parse(text);
    if (result.success) {
    // 標記已同步
    toSync.forEach(function (r) { markSynced(r.name); });
    debugInfo.textContent = '✅ 已同步 ' + (result.processed || toSync.length) + ' 筆 | 資料庫: ' + Object.keys(participantMap).length + ' 筆';
    } else {
    // 同步失敗 → 放回佇列
    syncQueue = toSync.concat(syncQueue);
    updateQueueBar();
    debugInfo.textContent = '⚠️ 同步失敗: ' + (result.message || '未知');
    }
    } catch (e) {
    // 可能是 GAS 重導向 HTML → 試著當作成功
    toSync.forEach(function (r) { markSynced(r.name); });
    debugInfo.textContent = '📡 已送出 ' + toSync.length + ' 筆';
    }
    })
    .catch(function (err) {
    isSyncing = false;
    // 網路失敗 → 放回佇列
    syncQueue = toSync.concat(syncQueue);
    updateQueueBar();
    debugInfo.textContent = '⚠️ 網路錯誤，待重試: ' + syncQueue.length + ' 筆';
    });
    }
    // ====== UI 輔助 ======
    function updateQueueBar() {
    var count = syncQueue.length;
    if (count > 0) {
    queueBarEl.className = 'queue-bar has-items';
    queueBarEl.textContent = '📡 待同步: ' + count + ' 筆 (每 ' + (SYNC_INTERVAL / 1000) + ' 秒自動上傳)';
    } else {
    queueBarEl.className = 'queue-bar';
    queueBarEl.textContent = '✅ 所有紀錄已同步';
    }
    }
    function addRecentScan(name, synced) {
    recentScans.unshift({ name: name, time: new Date().toLocaleTimeString('zh-TW', { hour: '2-digit', minute: '2-digit', second: '2-digit' }), synced: synced });
    if (recentScans.length > 20) recentScans.pop();
    renderRecentList();
    }
    function markSynced(name) {
    for (var i = 0; i < recentScans.length; i++) {
    if (recentScans[i].name === name && !recentScans[i].synced) {
    recentScans[i].synced = true;
    break;
    }
    }
    renderRecentList();
    }
    function renderRecentList() {
    var html = '';
    recentScans.forEach(function (r) {
    var syncIcon = r.synced
    ? '<span class="sync-icon sync-done">✓</span>'
    : '<span class="sync-icon sync-pending">⏳</span>';
    html += '<div class="recent-item"><span class="name">' + r.name + '</span><span class="time">' + r.time + syncIcon + '</span></div>';
    });
    recentListEl.innerHTML = html;
    }

    // ====== 相機 & 掃描 ======
    function startCamera() {
    if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
    showError('此瀏覽器不支援相機，請使用 Chrome 或 Safari。');
    return;
    }
    var constraints = {
    video: {
    facingMode: 'environment',
    width: { ideal: 640 },
    height: { ideal: 480 },
    focusMode: { ideal: 'continuous' },
    frameRate: { ideal: 15, max: 30 }
    },
    audio: false
    };
    navigator.mediaDevices.getUserMedia(constraints)
    .then(function (stream) {
    video.srcObject = stream;
    try {
    var track = stream.getVideoTracks()[0];
    var caps = track.getCapabilities ? track.getCapabilities() : {};
    if (caps.focusMode && caps.focusMode.indexOf('continuous') > -1) {
    track.applyConstraints({ advanced: [{ focusMode: 'continuous' }] });
    }
    } catch (e) { }
    video.play().then(function () {
    debugInfo.textContent = '相機: ' + video.videoWidth + 'x' + video.videoHeight + ' | 處理: ' + MAX_SCAN_WIDTH + 'px';
    setInterval(scanFrame, SCAN_INTERVAL);
    });
    })
    .catch(function (err) {
    if (err.name === 'OverconstrainedError') {
    navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' }, audio: false })
    .then(function (stream) {
    video.srcObject = stream;
    video.play().then(function () {
    setInterval(scanFrame, SCAN_INTERVAL);
    });
    })
    .catch(function (e2) { showError('相機啟動失敗：' + e2.message); });
    } else if (err.name === 'NotAllowedError') {
    showError('相機權限被拒絕。請允許此頁面使用相機，然後重新整理。');
    } else {
    showError('相機啟動失敗：' + err.message);
    }
    });
    }
    function scanFrame() {
    if (!scanning) return;
    if (video.readyState !== video.HAVE_ENOUGH_DATA) return;
    var vw = video.videoWidth, vh = video.videoHeight;
    if (!vw || !vh) return;
    var scale = Math.min(1, MAX_SCAN_WIDTH / vw);
    var sw = Math.floor(vw * scale);
    var sh = Math.floor(vh * scale);
    canvas.width = sw;
    canvas.height = sh;
    ctx.drawImage(video, 0, 0, sw, sh);
    var imageData;
    try { imageData = ctx.getImageData(0, 0, sw, sh); } catch (e) { return; }
    var code = jsQR(imageData.data, imageData.width, imageData.height, {
    inversionAttempts: 'attemptBoth'
    });
    scanCount++;
    if (code && code.data) {
    handleScanResult(code.data);
    }
    if (scanCount % 20 === 0) {
    var qLen = syncQueue.length;
    debugInfo.textContent = '處理: ' + sw + 'x' + sh + ' | 已掃 ' + scanCount + ' 幀 | 佇列: ' + qLen;
    }
    }
